@echo off
title VS13 Fix
color 0f
cls

echo Begin Cleanup and Fix? to skip cleanup phase just type fix

set /p var=(Y/N)

IF '%var%' == 'y' GOTO cleanup
IF '%var%' == 'Y' GOTO cleanup
IF '%var%' == 'n' GOTO exit
IF '%var%' == 'N' GOTO exit
IF '%var%' == 'fix' GOTO fix

:cleanup
cd C:\windows\temp
cls
echo Cleaning Logs (This may take a few minutes..)
del *.LOG /Q
echo ---------------------------------------------
echo Log cleanup Successful. Proceed to apply fix?

set /p var2=(Y/N)
IF '%var2%' == 'y' GOTO fix
IF '%var2%' == 'Y' GOTO fix
IF '%var2%' == 'n' GOTO exit
IF '%var2%' == 'N' GOTO exit

:fix
cd %SystemRoot%\Microsoft.NET\Framework
cls
echo Applying fix now..
mkdir URTInstall_GAC
mkdir URTInstallPath_GAC
echo If the fix didn't apply please relaunch the script as administrator and try again. Don't forget to type fix to skip cleanup.
pause
exit

GOTO exit
